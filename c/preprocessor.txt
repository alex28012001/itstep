директивы препроцессора начинаются с #
#[имя_директивы][ликсемы_препроцессора]

директивы препроцессора: 
include - включить содержимое файла
pragma - действие определяется компилятором
define - определение макроса
undef -отмена определения макроса
if -компиляция, если истина
ifdef - компиляция, если макрос определен
ifndef -компиляция, если макрос не опреденен
else - альтернатива 3 выше упомянутым
elif -else/if
endif - окончание группы компиляции по условию
error - прерывает компиляцию с выдачей сообщения


#define size 100  в программе заменяет все size на 100
#define max(a,b) ((a)>(b)?a:b) - определяет функцию

#define s(x) #x - делает переменную текстом

__DATE__  месяц день год -выводит дату
__FILE__ -расположение файла
__LINE__ номер строки
__TIME__ -выводит время
